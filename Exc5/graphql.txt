type Client {
    id: ID!
    name: String
    age: Int
    documents: [Document]
    relatives: [Relative]
}  

type Document {
    id: ID!
    type: String
    number: String
    issueDate: String
    expiryDate: String
}

type Relative {
    id: ID!
    relationType: String
    name: String
    age: Int
}


type Query {
    client(id: ID!): Client
    clientsDocuments(id: ID!): [Document]
    clientsRelatives(id: ID!): [Relative]
}


1. Клиент

query {
  client(id: "645") {
    id
    name
    age
    documents {
      id
      type
      number
      issueDate
      expiryDate
    }
    relatives {
      id
      name
      relationType
      age
    }
  }
}

2. Документ

query {
  clientsDocuments(id: "645") {
      id
      type
      number
      issueDate
      expiryDate
  }
}

3. Родственники

query {
  clientsRelatives(id: "645") {
      id
      name
      relationType
      age
  }
}
